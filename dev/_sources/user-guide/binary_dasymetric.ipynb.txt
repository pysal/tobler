   "source": [
    "def iterclip(source, mask):\n",
    "    tree = mask.sindex\n",
    "    features = []\n",
    "    for i, row in source.iterrows():\n",
    "        query = tree.query(source.geometry.iloc[i], predicate=\"intersects\").tolist()\n",
    "        submask = mask[mask.index.isin(query)]\n",
    "        single = gpd.GeoDataFrame(row.to_frame().T, crs=source.crs)\n",
    "        features.append(gpd.clip(single, submask))\n",
    "    frame = gpd.GeoDataFrame(pd.concat(features), crs=source.crs)\n",
    "    return frame"
   ]
  },
